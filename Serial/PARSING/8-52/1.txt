void setup() {
  // put your setup code here, to run once:
  Serial.begin(9600);
  Serial.setTimeout(1000);
  pinMode(3, OUTPUT);
  pinMode(4, OUTPUT);
  pinMode(5, OUTPUT);
}

void loop() {
  // put your main code here, to run repeatedly:
  if (Serial.available())
  {
    char data[30];
    int amount = Serial.readBytesUntil(';', data, 30);
    data[amount] = NULL;
    Serial.println(data);
  }

}
